```dart
// lib/models/station.dart

class StationFuel {
  final String name;
  final String type;
  final double price;

  const StationFuel({
    required this.name,
    required this.type,
    required this.price,
  });

  factory StationFuel.fromJson(Map<String, dynamic> json) => StationFuel(
        name: json['name'] as String,
        type: json['type'] as String,
        price: (json['price'] as num).toDouble(),
      );

  /// Sort priority for known fuel types. Unknown types sort to the end.
  static const _sortOrder = ['es95', 'es100', 'eurodizel', 'lpg', 'unp10kg'];

  int get sortIndex {
    final idx = _sortOrder.indexOf(type);
    return idx >= 0 ? idx : 999;
  }
}

class Station {
  final String id;
  final String name;
  final String url;
  final String updated;
  final List<StationFuel> fuels;

  const Station({
    required this.id,
    required this.name,
    required this.url,
    required this.updated,
    required this.fuels,
  });

  factory Station.fromJson(Map<String, dynamic> json) => Station(
        id: json['id'] as String,
        name: json['name'] as String,
        url: json['url'] as String,
        updated: json['updated'] as String,
        fuels: (json['fuels'] as List)
            .map((f) => StationFuel.fromJson(f as Map<String, dynamic>))
            .toList(),
      );

  /// Returns fuels sorted: known types first (es95, es100, eurodizel, lpg, unp10kg),
  /// then remaining alphabetically by name.
  List<StationFuel> get sortedFuels {
    final sorted = List<StationFuel>.from(fuels);
    sorted.sort((a, b) {
      final cmp = a.sortIndex.compareTo(b.sortIndex);
      if (cmp != 0) return cmp;
      return a.name.compareTo(b.name);
    });
    return sorted;
  }
}

class StationsResponse {
  final String updated;
  final List<Station> stations;

  const StationsResponse({required this.updated, required this.stations});

  factory StationsResponse.fromJson(Map<String, dynamic> json) =>
      StationsResponse(
        updated: json['updated'] as String,
        stations: (json['stations'] as List)
            .map((s) => Station.fromJson(s as Map<String, dynamic>))
            .toList(),
      );
}

/// Format ISO date string (YYYY-MM-DD) to Croatian format (DD.MM.YYYY.)
String formatDateCroatian(String dateStr) {
  try {
    final parts = dateStr.split('-');
    return '${parts[2]}.${parts[1]}.${parts[0]}.';
  } catch (_) {
    return dateStr;
  }
}
```

```dart
// lib/data/repositories/station_repository.dart
import '../database.dart';
import '../../models/station.dart';

class StationRepository {
  final AppDatabase db;

  StationRepository(this.db);

  /// Replace all station data with new response.
  Future<void> saveStations(StationsResponse response) async {
    final database = await db.database;
    await database.transaction((txn) async {
      await txn.delete('station_fuels');
      await txn.delete('stations');

      for (final station in response.stations) {
        await txn.insert('stations', {
          'id': station.id,
          'name': station.name,
          'url': station.url,
          'updated': station.updated,
        });
        for (final fuel in station.fuels) {
          await txn.insert('station_fuels', {
            'station_id': station.id,
            'name': fuel.name,
            'type': fuel.type,
            'price': fuel.price,
          });
        }
      }
    });
  }

  /// Get all cached stations with their fuels.
  Future<List<Station>> getStations() async {
    final stationRows = await db.query('stations', orderBy: 'name ASC');
    final stations = <Station>[];

    for (final row in stationRows) {
      final fuelRows = await db.query(
        'station_fuels',
        where: 'station_id = ?',
        whereArgs: [row['id']],
      );
      stations.add(Station(
        id: row['id'] as String,
        name: row['name'] as String,
        url: row['url'] as String,
        updated: row['updated'] as String,
        fuels: fuelRows
            .map((f) => StationFuel(
                  name: f['name'] as String,
                  type: f['type'] as String,
                  price: f['price'] as double,
                ))
            .toList(),
      ));
    }
    return stations;
  }

  /// Returns true if data should be re-fetched (never fetched or >24h ago).
  Future<bool> shouldFetch() async {
    final rows = await db.query('station_fetch_time');
    if (rows.isEmpty) return true;
    final lastFetch = DateTime.parse(rows.first['fetched_at'] as String);
    return DateTime.now().difference(lastFetch).inHours >= 24;
  }

  /// Record that a successful fetch just happened.
  Future<void> recordFetchTime() async {
    final now = DateTime.now().toIso8601String();
    final existing = await db.query('station_fetch_time');
    if (existing.isEmpty) {
      await db.insert('station_fetch_time', {'id': 1, 'fetched_at': now});
    } else {
      await db.update('station_fetch_time', {'fetched_at': now}, where: 'id = 1');
    }
  }
}
```

```dart
// lib/blocs/stations_cubit.dart
import 'package:bloc/bloc.dart';
import 'package:fuel_price_app/data/repositories/station_repository.dart';
import 'package:fuel_price_app/data/services/station_price_service.dart';
import 'package:fuel_price_app/models/station.dart';

class StationsState {
  final List<Station> stations;
  final bool isLoading;
  final bool hasError;

  const StationsState({
    this.stations = const [],
    this.isLoading = false,
    this.hasError = false,
  });

  StationsState copyWith({
    List<Station>? stations,
    bool? isLoading,
    bool? hasError,
  }) =>
      StationsState(
        stations: stations ?? this.stations,
        isLoading: isLoading ?? this.isLoading,
        hasError: hasError ?? this.hasError,
      );
}

class StationsCubit extends Cubit<StationsState> {
  final StationPriceService service;
  final StationRepository repository;
  bool _loaded = false;

  StationsCubit({
    required this.service,
    required this.repository,
  }) : super(const StationsState());

  /// Load station data. Fetches from network if needed, otherwise uses cache.
  Future<void> load() async {
    emit(state.copyWith(isLoading: true, hasError: false));

    try {
      final shouldFetch = await repository.shouldFetch();

      if (shouldFetch) {
        final response = await service.fetchStations();
        if (response != null) {
          await repository.saveStations(response);
          await repository.recordFetchTime();
        }
      }

      final stations = await repository.getStations();
      _loaded = true;
      emit(state.copyWith(isLoading: false, stations: stations, hasError: stations.isEmpty));
    } catch (_) {
      emit(state.copyWith(isLoading: false, hasError: true));
    }
  }

  /// Force refresh — always fetch from network.
  Future<void> refresh() async {
    emit(state.copyWith(isLoading: true, hasError: false));

    try {
      final response = await service.fetchStations();
      if (response != null) {
        await repository.saveStations(response);
        await repository.recordFetchTime();
      }

      final stations = await repository.getStations();
      emit(state.copyWith(isLoading: false, stations: stations, hasError: stations.isEmpty));
    } catch (_) {
      emit(state.copyWith(isLoading: false, hasError: true));
    }
  }
}
```

```python
                    fuels.append({"name": name, "type": fuel_type, "price": price})

        if not fuels:
            print("Shell: no prices found", file=sys.stderr)
            return None

        return {
            "id": "shell",
            "name": "Shell",
            "url": "https://www.shell.hr/motoristi/shell-goriva/cijena-goriva.html",
            "updated": datetime.now(timezone.utc).strftime("%Y-%m-%d"),
            "fuels": fuels,
        }
    except Exception as e:
        print(f"Shell scrape failed: {e}", file=sys.stderr)
        return None


def _parse_price(text: str) -> float | None:
    """Parse a price string like '1,45', '1.45 €/L', etc."""
    import re
    # Remove currency symbols, units, whitespace
    cleaned = re.sub(r"[€/LlKkGg\s]", "", text.strip())
    # Handle comma as decimal separator
    cleaned = cleaned.replace(",", ".")
    try:
        price = float(cleaned)
        # Sanity check: fuel prices should be between 0.01 and 50 EUR
        if 0.01 <= price <= 50.0:
            return round(price, 2)
        return None
    except ValueError:
        return None


def _classify_fuel(name: str) -> str:
    """Classify a fuel name into a standardized type key."""
    lower = name.lower()

    if "100" in lower and ("super" in lower or "euro" in lower or "benzin" in lower):
        return "es100"
    if "95" in lower and ("super" in lower or "euro" in lower or "benzin" in lower):
        return "es95"
    if "lpg" in lower or "autoplin" in lower or "auto plin" in lower:
        return "lpg"
    if "unp" in lower or "boca" in lower or "butan" in lower:
        return "unp10kg"
    # Premium diesel variants
    if ("blue" in lower or "premium" in lower or "v-power" in lower or "optima" in lower) and (
        "dizel" in lower or "diesel" in lower
    ):
        return "premium_diesel"
    # Premium 95/98 variants
    if ("blue" in lower or "premium" in lower or "v-power" in lower or "optima" in lower) and (
        "95" in lower or "98" in lower or "super" in lower or "benzin" in lower
    ):
        return "premium_95"
    if "dizel" in lower or "diesel" in lower:
        return "eurodizel"

    return "other"


def main():
    scrapers = [scrape_ina, scrape_tifon, scrape_crodux, scrape_petrol, scrape_shell]

    stations = []
    for scraper in scrapers:
        result = scraper()
        if result is not None:

    OUTPUT_PATH.write_text(json.dumps(output, indent=2, ensure_ascii=False) + "\n")
    print(f"\nWrote {len(stations)} stations to {OUTPUT_PATH}")


if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""Check Narodne Novine for new fuel price regulations and update fuel_params.json."""

import json
import sys
from pathlib import Path

import requests
from bs4 import BeautifulSoup

CONFIG_PATH = Path(__file__).parent.parent / "config" / "fuel_params.json"

HEADERS = {
    "User-Agent": "Mozilla/5.0 (compatible; FuelLens/2.0; +https://github.com/jurajnakic/fuel-price-app)"
}

# Search for fuel price regulation decrees on NN
NN_SEARCH_URL = "https://narodne-novine.nn.hr/search.aspx"


def load_current_config() -> dict:
    """Load the current fuel_params.json."""
    return json.loads(CONFIG_PATH.read_text(encoding="utf-8"))


def check_for_new_regulations() -> dict | None:
    """
    Check Narodne Novine for new fuel price regulations.

    Searches for "uredba cijene naftnih derivata" and checks if there are
    newer regulations than the one currently in fuel_params.json.

    Returns the regulation info dict if a new one is found, None otherwise.
    """
    try:
        current = load_current_config()
        current_nn = current["price_regulation"]["nn_reference"]
        print(f"Current regulation: {current_nn}")

        # Search NN for fuel price regulations
        params = {
            "query": "uredba utvrđivanju najviših maloprodajnih cijena naftnih derivata",
            "sort": "date",
            "order": "desc",
        }
        resp = requests.get(NN_SEARCH_URL, params=params, headers=HEADERS, timeout=15)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, "html.parser")

        # Look for regulation links in search results
        results = soup.select(".search-result a, .result-item a, h3 a")
        for link in results:
            text = link.get_text(strip=True)
            href = link.get("href", "")

            # Check if this looks like a fuel price regulation
            if "cijena" in text.lower() and "naft" in text.lower():
                # Extract NN reference (e.g., "NN 31/2025")
                import re
                nn_match = re.search(r"NN\s*(\d+/\d{4})", text)
                if nn_match:
                    nn_ref = f"NN {nn_match.group(1)}"
                    if nn_ref != current_nn:
                        print(f"Found new regulation: {nn_ref}")
                        return {
                            "nn_reference": nn_ref,
                            "url": href if href.startswith("http") else f"https://narodne-novine.nn.hr{href}",
                            "text": text,
                        }

        print("No new regulations found.")
        return None

    except Exception as e:
        print(f"Regulation check failed: {e}", file=sys.stderr)
        return None


def main():
    new_reg = check_for_new_regulations()

    if new_reg is None:
        print("No update needed.")
        return

    # We found a new regulation — update the reference info but NOT the numeric parameters.
    # Numeric parameters (premiums, excise, density) require manual verification.
    current = load_current_config()
    current["price_regulation"]["nn_reference"] = new_reg["nn_reference"]
    if new_reg.get("url"):
        current["price_regulation"]["nn_url"] = new_reg["url"]

    CONFIG_PATH.write_text(json.dumps(current, indent=2, ensure_ascii=False) + "\n", encoding="utf-8")
    print(f"\nUpdated regulation reference to {new_reg['nn_reference']}")
    print("⚠️  IMPORTANT: Numeric parameters (premiums, excise, density) were NOT updated.")
    print("    Manual review required to verify and update formula parameters.")


if __name__ == "__main__":
    main()
```
